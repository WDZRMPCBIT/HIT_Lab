Program -> Defination Program | Assignment Program | Control Program | Function Program | Struct Program | epsilon

Struct -> struct id { Statement }
Statement -> Defination | Defination , Statement

Defination -> Data id ;
Data -> Type Array | id Array
Type -> int | float | bool
Array -> [ const ] Array | epsilon

Assignment -> id = Value ; | id Index = Value ;
Value -> Value + Value | Value - Value | Value * Value | Value / Value
Value -> - Value
Value -> ( Value )
Value -> const | id Index
Index -> epsilon | [ Value ] Index

Control -> If | If Else | While
If -> if ( Condition ) { Program }
Else -> else { Program }
While -> while ( Condition ) { Program }
Condition -> Condition and Condition | Condition or Condition | not Condition | ( Condition ) | Value Relop Value | 1 | 0
Relop -> < | <= | == | != | > | >=

Function -> Data id ( Parameter ) { Process }
Parameter -> epsilon | Defination | Defination , Parameter
Process -> Program Return
Return -> epsilon | return id Index ;

end

L -> * R
L -> id
R -> L

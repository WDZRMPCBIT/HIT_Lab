Program -> Line | Line Program
Line -> Defination | Assignment | Control | Function | Struct | Call ;

Struct -> struct id { Statement }
Statement -> Defination | Defination Statement

Defination -> Variable ;
Variable -> Data id | Data Array id
Data -> Type | id
Type -> int | float | bool
Array -> [ const ] Array

Assignment -> id = Value ; | id Index = Value ;
Value -> Value + Value | Value - Value | Value * Value | Value / Value | Call
Value -> - Value
Value -> ( Value )
Value -> const | id | id Index
Index -> [ Value ] Index

Control -> If | If Else | While
If -> if ( Condition ) { Program }
Else -> else { Program }
While -> while ( Condition ) { Program }
Condition -> Condition and Condition | Condition or Condition | not Condition | ( Condition ) | Value Relop Value | 1 | 0
Relop -> < | <= | == | != | > | >=

Function -> Data id ( Parameter ) { Process } | Data id ( ) { Process }
Parameter -> Variable | Variable , Parameter
Process -> Program | Program Return
Return -> return id Index ; | return id ;

Call -> id ( Transmit )
Transmit -> id | id , Transmit | id Index | id Index , Transmit

end

S -> L = R
S -> R
L -> * R
L -> id
R -> L

end
